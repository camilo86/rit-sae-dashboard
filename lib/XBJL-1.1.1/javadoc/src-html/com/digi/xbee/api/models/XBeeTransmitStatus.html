012package com.digi.xbee.api.models;
013
014import java.util.HashMap;
015
016/**
017 * Enumerates the different transmit status. Transmit status field is part of 
018 * the {@code TransmitStatusPacket} and {@code TXStatusPacket} indicating the 
019 * status of the transmission.
020 * 
021 * @see com.digi.xbee.api.packet.common.TransmitStatusPacket
022 * @see com.digi.xbee.api.packet.raw.TXStatusPacket
023 */
024public enum XBeeTransmitStatus {
025
026        // Enumeration types
027        SUCCESS (0x00, "Success"),
028        NO_ACK (0x01, "No acknowledgement received"),
029        CCA_FAILURE (0x02, "CCA failure"),
030        PURGED (0x03, "Transmission purged, it was attempted before stack was up"),
031        WIFI_PHYSICAL_ERROR (0x04, "Physical error occurred on the interface with the WiFi transceiver"),
032        INVALID_DESTINATION (0x15, "Invalid destination endpoint"),
033        NO_BUFFERS (0x18, "No buffers"),
034        NETWORK_ACK_FAILURE (0x21, "Network ACK Failure"),
035        NOT_JOINED_NETWORK (0x22, "Not joined to network"),
036        SELF_ADDRESSED (0x23, "Self-addressed"),
037        ADDRESS_NOT_FOUND (0x24, "Address not found"),
038        ROUTE_NOT_FOUND (0x25, "Route not found"),
039        BROADCAST_FAILED (0x26, "Broadcast source failed to hear a neighbor relay the message"),
040        INVALID_BINDING_TABLE_INDEX (0x2B, "Invalid binding table index"),
041        INVALID_ENDPOINT (0x2C, "Invalid endpoint"),
042        BROADCAST_ERROR_APS (0x2D, "Attempted broadcast with APS transmission"),
043        BROADCAST_ERROR_APS_EE0 (0x2E, "Attempted broadcast with APS transmission, but EE=0"),
044        SOFTWARE_ERROR (0x31, "A software error occurred"),
045        RESOURCE_ERROR (0x32, "Resource error lack of free buffers, timers, etc."),
046        PAYLOAD_TOO_LARGE (0x74, "Data payload too large"),
047        INDIRECT_MESSAGE_UNREQUESTED (0x75, "Indirect message unrequested"),
048        SOCKET_CREATION_FAILED (0x76, "Attempt to create a client socket failed"),
049        IP_PORT_NOT_EXIST (0x77, "TCP connection to given IP address and port doesn't exist. Source port is non-zero so that a new connection is not attempted"),
050        UDP_SRC_PORT_NOT_MATCH_LISTENING_PORT (0x78, "Source port on a UDP transmission doesn't match a listening port on the transmitting module."),
051        KEY_NOT_AUTHORIZED (0xBB, "Key not authorized"),
052        UNKNOWN (255, "Unknown");
053        
054        // Variables
055        private final int id;
056        
057        private final String description;
058        
059        private static final HashMap<Integer, XBeeTransmitStatus> lookupTable = new HashMap<Integer, XBeeTransmitStatus>();
060        
061        static {
062                for (XBeeTransmitStatus ts:values())
063                        lookupTable.put(ts.getId(), ts);
064        }
065        
066        /**
067         * Class constructor. Instantiates a new {@code XBeeTransmitStatus} entry 
068         * with the given parameters.
069         * 
070         * @param id Transmit status identifier.
071         * @param description Transmit status description.
072         */
073        private XBeeTransmitStatus(int id, String description) {
074                this.id = id;
075                this.description = description;
076        }
077        
078        /**
079         * Returns the numeric value of the transmit status identifier.
080         * 
081         * @return Transmit status identifier.
082         */
083        public int getId() {
084                return id;
085        }
086        
087        /**
088         * Returns the transmit status description.
089         * 
090         * @return Transmit status description.
091         */
092        public String getDescription() {
093                return description;
094        }
095        
096        /**
097         * Returns the {@code XBeeTransmitStatus} associated to the given 
098         * identifier. 
099         * 
100         * Returns {@code UNKNOWN} if the identifier is not in the enumeration.
101         * 
102         * @param id Identifier of the {@code XBeeTransmitStatus} to retrieve.
103         * 
104         * @return The {@code XBeeTransmitStatus} associated to the given 
105         *         identifier.
106         *         {@code UNKNOWN} if the identifier is not in the enumeration.
107         */
108        public static XBeeTransmitStatus get(int id) {
109                XBeeTransmitStatus status = lookupTable.get(id);
110                if (status != null)
111                        return status;
112                return UNKNOWN;
113        }
114        
115        /*
116         * (non-Javadoc)
117         * @see java.lang.Enum#toString()
118         */
119        @Override
120        public String toString() {
121                if (id != SUCCESS.getId())
122                        return String.format("Error: %s (0x%02X)", description, id);
123                else
124                        return String.format("%s (0x%02X)", description, id);
125        }
126}
